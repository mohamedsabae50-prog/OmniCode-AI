async function runDebugger(isFollowUp = false) { const selectedLang = document.getElementById('progLang').value.toLowerCase(); document.getElementById('loader').classList.remove('hidden'); const fd = new FormData(); fd.append('code', isFollowUp ? document.querySelector('#preCode code').innerText : document.getElementById('codeInput').value); fd.append('lang', selectedLang); fd.append('ui_lang', document.getElementById('uiLang').value); fd.append('inquiry', document.getElementById('inquiryInput').value || "Fix"); fd.append('error_log', document.getElementById('errorInput').value || ""); if(isFollowUp) fd.append('follow_up', document.getElementById('followUpInput').value); try { const res = await fetch('/api/index', { method: 'POST', body: fd }); const data = await res.json(); document.getElementById('expText').innerText = data.explanation; document.getElementById('complexityText').innerText = data.complexity || "N/A"; // تجهيز حاويات الكود const oldPre = document.getElementById('oldCode'); const newPre = document.getElementById('preCode'); // تنظيف الكلاسات القديمة تماماً وإضافة الكلاس الجديد oldPre.innerHTML = ``; newPre.innerHTML = ``; // وضع النص oldPre.querySelector('code').textContent = isFollowUp ? oldPre.querySelector('code').textContent : document.getElementById('codeInput').value; newPre.querySelector('code').textContent = data.result; // إجبار Prism على إعادة التلوين Prism.highlightElement(oldPre.querySelector('code')); Prism.highlightElement(newPre.querySelector('code')); document.getElementById('loader').classList.add('hidden'); document.getElementById('resultArea').classList.remove('hidden'); document.getElementById('placeholder').classList.add('hidden'); } catch (e) { document.getElementById('loader').classList.add('hidden'); alert("Connection Error"); } }
